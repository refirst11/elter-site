# Style

### Style.create
```ts title="style.css.ts"
import { Style } from 'typedcssx'

export const styles = Style.create({
  container: {
    height: 100,
    width: '100%',
    padding: '32px 0',
    display: 'flex',
    justifyContent: 'space-between'
  },
  link: {
    fontSize: '14px',
    textDecoration: 'none'
  },
})
```

### Style.global
```ts title="global.css.ts"
Style.global({
  body: {
    margin: 'auto',
    textAlign: 'center',
    lineHeight: 1.75,
    color: 'var(--color-text)',
    background: 'var(--color-bg)'
  },
  h1: {
    color: 'var(--color-heading)',
    padding: '20px 0',
    margin: '20px 0',
    fontWeight: '600'
  },
  h2: {
    color: 'var(--color-heading)',
    fontSize: '30px',
    fontWeight: '550',
  },
});
  ```

### Style.set

```ts title="style.css.ts"
import { Style } from 'typedcssx'

export const linkStyle = Style.set({
  fontSize: '14px',
  color: 'var(--color-link)',
  textDecoration: 'none'
})

export const boxStyle = Style.set({
  display: 'block',
  padding: 20,
  width: 60,
  height: 60
})

export const containerStyle = Style.set({
  display: 'flex',
  justifyContent: 'center',
  alignItems: 'center'
})
```
